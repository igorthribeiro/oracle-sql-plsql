/*
1 - MONTE UM RELATÓRIO QUE MOSTRE A SEGUINTE FRASE:

King ganha R$24.000,00 por mês mas gostaria de ganhar R$ 72.000,00(3x)

*/
SELECT LAST_NAME|| ' ganha '|| TO_CHAR(SALARY, 'fmL999G990D00')
       || ' por mês mas gostaria de ganhar ' || TO_CHAR(SALARY * 3, 'fmL999G990D00')
  FROM EMPLOYEES;
  
/*
2 - MONTE UM RELATÓRIO QUE MOSTRE A DATA DE CONTRATAÇÃO DO EMPREGADO
NO SEGUINTE FORMATO:

LAST_NAME  |  HIRE_DATE  | CONTRATADO EM:
King       |  17/09/2018 | segunda-feira, 17 de Setembro de 2018  
*/

SELECT LAST_NAME, HIRE_DATE, TO_CHAR(HIRE_DATE,'fmday, DD "de" Month "de" YYYY')
  FROM EMPLOYEES;

SELECT * FROM V$NLS_PARAMETERS WHERE PARAMETER = 'NLS_DATE_FORMAT';

ALTER SESSION SET NLS_DATE_FORMAT='DD/MM/YYYY - HH24:MI:SS';
  
SELECT NEXT_DAY(SYSDATE, 'TERÇA') FROM DUAL; 
SELECT NEXT_DAY(SYSDATE, 3) FROM DUAL; 

/*
3 - DESAFIO: TODO EMPREGADO PRECISA FAZER UM EXAME NA PRIMEIRA
SEGUNDA FEIRA APÓS 6 MESES DE SUA CONTRATAÇÃO.
MONTE UM RELATÓRIO QUE MOSTRE O EMPLOYEE_ID, LAST_NAME, HIRE_DATE
E A DATA DO EXAME DO EMPREGADO POR EXTENSO:

ID    LNAME  HDATE        DATA EXAME
100 - KING - 17/01/2001 - 23 DE JULHO DE 2001
*/
ALTER SESSION SET NLS_DATE_FORMAT='DD/MM/RR';
SELECT EMPLOYEE_ID, LAST_NAME, HIRE_DATE,
       TO_CHAR(
       NEXT_DAY(ADD_MONTHS(HIRE_DATE,6),2),
       'fmDD "DE" MONTH "DE" YYYY') DATA_EXAME
  FROM EMPLOYEES;
