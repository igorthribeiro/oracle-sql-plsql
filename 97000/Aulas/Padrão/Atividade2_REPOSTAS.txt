/*
1- VAMOS CRIAR UM LOGIN PARA OS EMPREGADOS DO RH. 
A FORMULA DO LOGIN EH:
- #
- AS 3 PRIMEIRAS LETRAS DO SEU NOME
- AS 2 ULTIMAS LETRAS DO SEU SOBRENOME
- COLOCAR TUDO EM LETRA MINÚSCULA
Exibir: EMPLOYEE_ID, FIRST_NAME, LAST_NAME E LOGIN

Ex.: 
300 - IGOR - RIBEIRO - #igoro
*/
SELECT EMPLOYEE_ID, FIRST_NAME, LAST_NAME,
       LOWER('#'||SUBSTR(FIRST_NAME,1,3)||SUBSTR(LAST_NAME,-2)) LOGIN,
       LOWER(CONCAT('#',CONCAT(SUBSTR(FIRST_NAME,1,3),SUBSTR(LAST_NAME,-2)))) LOGIN2
  FROM EMPLOYEES;
  
/*
2 - OS USUÁRIOS DO RH TERÃO UM DESCONTO DE 0,0753% EM SEU SALARIO
PARA FUNDO DE POUPANÇA PESSOAL.
MOSTRE O EMPLOYEE_ID, FIRST_NAME, SALARY E O VALOR DO DESCONTO
ARREDONDADO PARA 1 DECIMAIS:
Ex.:
300 - IGOR - 1550 - 1.2
*/
SELECT EMPLOYEE_ID, FIRST_NAME, SALARY, ROUND(SALARY * 0.000753, 2) DESCONTO
  FROM EMPLOYEES;

/*
3 - DESAFIO: CRIE UMA QUERY QUE MOSTRE O FIRST_NAME, LAST_NAME,
O FIRST_NAME CONCATENADO COM LAST_NAME E A QUANTIDADE DE *
CORRESPONDENTE AO TAMANHO DO FIRST_NAME CONCATENADO COM LAST_NAME.

Ex.:
IGOR - RIBEIRO - IGORRIBEIRO - ***********
*/

SELECT FIRST_NAME,
       LAST_NAME,
       CONCAT(FIRST_NAME, LAST_NAME) NOMESOBRENOME,
       LPAD('*',LENGTH(CONCAT(FIRST_NAME, LAST_NAME)),'*')
  FROM EMPLOYEES;
