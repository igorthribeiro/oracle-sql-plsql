/*
1 - O DEPARTAMENTO DE RH PRECISA DE UM RELATÓRIO QUE MOSTRE O NOME
E O SOBRENOME DO EMPREGADO E A SUA FUNÇÃO SEPARADOS POR VÍRGULA:

FIRST_NAME/LAST_NAME/JOB_ID
*/

SELECT FIRST_NAME||' '||LAST_NAME||', '||JOB_ID "NOME,JOB"
  FROM EMPLOYEES;
  
/*
2 - MONTE UM RELATÓRIO QUE MOSTRE O EMPLOYEE_ID, FIRST_NAME E SALARY
DOS EMPREGADOS QUE GANHAM ENTRE $5000 E $12000. ORDENE O RESULTADO
PELO FIRST_NAME E COLOQUE APELIDO PARA AS COLUNAS EM PORTUGUÊS:
SALARY --> SALARIO.
*/
  
SELECT EMPLOYEE_ID "#ID",
       FIRST_NAME NOME,
       SALARY SALARIO
  FROM EMPLOYEES
 WHERE SALARY BETWEEN 5000 AND 12000
 ORDER BY FIRST_NAME;
/*
3 - DESAFIO: CRIE UM QUERY QUE MOSTRE O EMPLOYEE_ID, FIRST_NAME,
SALARY, SALARY * 15% (AUMENTO) E A UMA COLUNA MOSTRE A DIFERENÇA
ENTRE O AUMENTO E O SALÁRIO ATUAL:
*/
SELECT EMPLOYEE_ID ID, FIRST_NAME F_NAME,
       SALARY SALARIO, SALARY * 1.15 AUMENTO, SALARY * 1.15 - SALARY DIFERENÇA
  FROM EMPLOYEES;
