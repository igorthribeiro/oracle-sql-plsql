/*
1 - MONTRE UM QUERY QUE MOSTRE O ID DE TODOS OS DEPARTAMENTOS
QUE NÃO POSSUEM EMPREGADO SA_REP UTILIZANDO MINUS.

2 - DESAFIO: VOCÊ CONSEGUE MOSTRAR O NOME DOS DEPARTAMENTOS
DA ATIVIDADE 1?
*/
SELECT D.DEPARTMENT_ID, D.DEPARTMENT_NAME
  FROM DEPARTMENTS D JOIN (SELECT DEPARTMENT_ID FROM DEPARTMENTS
                           MINUS
                           SELECT DEPARTMENT_ID FROM EMPLOYEES
                            WHERE JOB_ID = 'SA_REP') DP
                       ON D.DEPARTMENT_ID = DP.DEPARTMENT_ID;
