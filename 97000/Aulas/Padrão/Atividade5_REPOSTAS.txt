/*
1 - CONSTRUA UMA QUERY QUE MOSTRE:
DEPARTMENT_ID, DEPARTMENT_NAME, CITY, COUNTRY_NAME, REGION_NAME,
EMPLOYEE_ID, FIRST_NAME, JOB_TITLE, SALARY
DOS EMPREGADOS DO ORA1.
*/

SELECT D.DEPARTMENT_ID, D.DEPARTMENT_NAME, L.CITY, C.COUNTRY_NAME,
       R.REGION_NAME, E.EMPLOYEE_ID, E.FIRST_NAME, J.JOB_TITLE,
       E.SALARY
  FROM DEPARTMENTS D JOIN LOCATIONS L
       ON D.LOCATION_ID = L.LOCATION_ID
  JOIN COUNTRIES C
       ON C.COUNTRY_ID = L.COUNTRY_ID
  JOIN REGIONS R
       ON R.REGION_ID = C.REGION_ID
  RIGHT JOIN EMPLOYEES E
       ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
  JOIN JOBS J
       ON J.JOB_ID = E.JOB_ID;
